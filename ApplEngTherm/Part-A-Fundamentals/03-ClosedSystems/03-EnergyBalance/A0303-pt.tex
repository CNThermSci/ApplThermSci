\subsection{Primeira Lei da Termodinâmica}
%-----------------------------------------------------------------------------------------------

    % !j 96 -i8
    %-------------------------------------------------------------------------------------------
    \begin{frame}{Enunciado}\vspace*{-2em}
        \uncover<1->{A \alert{1$^{\mathrm{a}}$ lei da Termodinâmica} estabelece que:}
        \begin{itemize}
            \item<1-> \alert{Energia} é uma quantidade \alert{conservada}. \\[\bigskipamount]
        \end{itemize}
        \uncover<2->{Este \alert{princípio da conservação da energia}:}
        \begin{itemize}
            \item<2-> É exaustivamente \alert{confirmado} em experimentos.%\\[\bigskipamount]
        \end{itemize}
    \end{frame}
    %-------------------------------------------------------------------------------------------

    % !j 96 -i8
    %-------------------------------------------------------------------------------------------
    \begin{frame}{Algumas Implicações}\vspace*{-2em}
        \uncover<1->{Logo, no universo observável:}
        \begin{itemize}
            \item<1-> \alert{Não há processos físicos} que \alert{criem} energia,
                \\[\medskipamount]
            \item<2-> \alert{Nem processos físicos} que \alert{destruam} energia.
                \\[\medskipamount]
            \item<3-> \alert{Processos físicos} podem apenas \alert{converter} energia de uma
                forma a outra. \\[\medskipamount]
        \end{itemize}
        \uncover<4->{A Relatividade Especial de Einstein:}
        \begin{itemize}
            \item<4-> \alert{Unificou} as conservações de \alert{massa} e de \alert{energia};
                \\[\medskipamount]
            \item<5-> Através da \alert{equivalência massa-energia} expressa por $E_{eq} =
                c^2m$.  \\[\medskipamount]
            \item<6-> Assim, a quantidade $E_{tot} = c^2m + E_{outras}$ do universo é
                \alert{conservada}. %\\[\medskipamount]
        \end{itemize}
    \end{frame}
    %-------------------------------------------------------------------------------------------

    % !j 96 -i8
    %-------------------------------------------------------------------------------------------
    \begin{frame}{Aplicações}\vspace*{-2em}
        \begin{columns}
        \column{0.50\textwidth}
        \uncover<1->{A 1$^{\mathrm{a}}$ lei é \alert{central} em Termodinâmica.\par\noindent
            Suas aplicações são \alert{vastas} e incluem:}
        \begin{itemize}
            \item<1-> Princípio em variedade de \alert{deduções}; \\[\medskipamount]
            \item<2-> Instrumental na \alert{definição} de \alert{propriedades}.
                \\[\medskipamount]
            \item<3-> Cálculos de \alert{processos} energéticos. %\\[\medskipamount]
        \end{itemize}
        \column{0.50\textwidth}
        \uncover<4->{Exemplo: O que é, afinal, ``energia''?}
        \begin{itemize}
            \item<6-> ``Energia é uma \alert{quantidade} (escalar) \\[\medskipamount]
            \item<7-> \hspace{1ex}que é \alert{conservada} na natureza \\[\medskipamount]
            \item<8-> \hspace{1ex}e que possui \alert{unidades de
                \kilogram\usk\squaren\meter\per\squaren\second}.'' %\\[\medskipamount]
        \end{itemize}
        \uncover<5->{\hfill --- Jack P.~Holman (SMU)}
        % https://www.smu.edu/News/2013/jack-holman-06may2013
        \end{columns}

\subsection{Balanço de Energia}
%-----------------------------------------------------------------------------------------------

    % !j 96 -i8
    %-------------------------------------------------------------------------------------------
    \begin{frame}{Balanço de Energia}\vspace*{-2em}
        \uncover<1->{A 1$^{\mathrm{a}}$ lei é matematicamente expressa por meio de
        \alert{balanço de energia}.}\\[\medskipamount]
        \uncover<2->{Em um \alert{processo}, o balanço de energia é dado por:}%
        \begin{equation*}%
            \uncover<3->{\left(\begin{array}{c}
                \text{Total de energia que} \\
                \text{\alert{entra} no sistema}
            \end{array}\right)-}
            \uncover<4->{\left(\begin{array}{c}
                \text{Total de energia} \\
                \text{que \alert{sai} do sistema}
            \end{array}\right)=}
            \uncover<5->{\left(\begin{array}{c}
                \text{Variação \alert{líquida} de} \\
                \text{energia no sistema}
            \end{array}\right),}%
        \end{equation*}%
        \uncover<6->{que matematicamente se escreve:}%
        \begin{equation*}%
            \uncover<6->{\alert{E_{ent} - E_{sai} = \Delta{E}_{sist} = E_2 -
            E_1},\quad\mbox{para um processo $1$--$2$.}}%
        \end{equation*}%
        \uncover<7->{Assim, se $E_1$, $E_{ent}$ e $E_{sai}$ são conhecidos, então: $E_2 = E_1 +
        E_{ent} - E_{sai}$.}
    \end{frame}

    \begin{frame}{Balanço de Energia -- Formas}\vspace*{-2em}
        \small
        \begin{align*}
            % --- `zeroth' line
            \uncover<1->{ & \text{Processo}} &
            \uncover<2->{
                \stackrel{\scriptstyle d()}{\longrightarrow}
            } & &
            \uncover<3->{\text{Diferenc}&\text{ial}} &
            \uncover<4->{
                \stackrel{\scriptstyle /dt}{\longrightarrow}
            } & &
            \uncover<5->{\text{Taxa} & }
            \\[\bigskipamount]
            % --- first line
            \uncover<1->{
                \alt<1,2,6,12>%
                {\alert{E_{ent} - E_{sai}} & \alert{= \Delta{E}_{sist} = E_2 - E_1}}%
                {E_{ent} - E_{sai} & = \Delta{E}_{sist} = E_2 - E_1}%
            } &
            \uncover<2->{
                \stackrel{\scriptstyle d()}{\longrightarrow}
            } & &
            \uncover<3->{
                \alt<3,4,8,12>%
                {\alert{\delta{E}_{ent} - \delta{E}_{sai}} & \alert{= dE_{sist}}}%

            } &
            \uncover<4->{
                \stackrel{\scriptstyle /dt}{\longrightarrow}
            } & &

    \begin{frame}{Balanço de Energia -- $E_{ent}$, $E_{sai}$}\vspace*{-2em}
        \uncover<1->{Em \alert{sistemas compressíveis simples}, $E_{ent}$ e $E_{sai}$ podem ser
        apenas nas formas de:}\\[\medskipamount]
        \begin{enumerate}
            \item<2-> \alert{calor} e \\[\medskipamount]
            \item<3-> \alert{trabalho}. \\[\bigskipamount]
        \end{enumerate}
        \uncover<4->{Assim, no balanço de energia:}
        \begin{align*}%
            \uncover<5->{
                \alert{E_{ent}} & \alert{= Q_{ent} + W_{ent}},\qquad\text{e}
            } \\[\medskipamount]
            \uncover<6->{
                \alert{E_{sai}} & \alert{= Q_{sai} + W_{sai}}.
            } %\\[\medskipamount]
        \end{align*}%
    \end{frame}
    %-------------------------------------------------------------------------------------------

    % !j 96 -i8
    %-------------------------------------------------------------------------------------------
    \begin{frame}{Balanço de Energia -- $E_{sist}$}\vspace*{-2em}
        \uncover<1->{Em \alert{sistemas clássicos} (pré-relativísticos) \alert{não reativos},
        $E_{sist} = E_{micro} + E_{macro}$:}\\[\medskipamount]
        \begin{enumerate}
            \item<2-> $E_{micro} \equiv U_{sist}$, a \alert{energia interna}, em \kilo\joule;
                \\[\medskipamount]
            \item<3-> $E_{macro} = E_c + E_p$, a soma das energias \alert{cinética}, e
                \alert{potencial}, em \kilo\joule, onde: \\[\medskipamount]
            \item<4-> $E_c = me_c = m\mathbb{V}^2\!/2$, com $\llbracket\mathbb{V}\rrbracket =
                \sqrt{\kilo\joule\per\kilogram} = \unit{\sqrt{1000}}{\meter\per\second} \approx
                \unit{31.6}{\meter\per\second} \approx \unit{114}{\kilo\meter\per\hour}$, ou
                \\[\medskipamount]
            \item<5-> $E_c = me_c = m\smcpbb{V}^2\!/2000$, com $\llbracket\smcpbb{V}\rrbracket =
                \meter\per\second = \sqrt{\joule\per\kilogram}$; \\[\medskipamount]
            \item<6-> $E_p = me_p = mg\mathbb{Z}$, com $\llbracket g\rrbracket =
                \meter\per\squaren\second$, $\llbracket\mathbb{Z}\rrbracket = \kilo\meter$ e
                $\llbracket g\mathbb{Z}\rrbracket = \kilo(\meter\per\second)^2 =
                \kilo(\joule\per\kilogram)$, ou \\[\medskipamount]
            \item<7-> $E_p = me_p = mg\smcpbb{Z}/1000$, com $\llbracket g\rrbracket =
                \meter\per\squaren\second$ e $\llbracket\smcpbb{Z}\rrbracket = \meter$.
                %\\[\medskipamount]
        \end{enumerate}
    \end{frame}
    %-------------------------------------------------------------------------------------------

        \begin{align*}
            % --- first line
            \uncover<1->{
                E_{ent} - E_{sai} &
                = \Delta{E}_{sist} = E_2 - E_1 \rightharpoondown
            } \\[\medskipamount]

%            } \\[\medskipamount]
            \uncover<2->{
                (Q_{ent} + W_{ent}) - (Q_{sai} + W_{sai}) &
                = \Delta{E}_{sist} = (U_2 + E_{c,2} + E_{p,2}) - (U_1 + E_{c,1} + E_{p,1})
                \rightharpoondown
            } \\[\medskipamount]
            \uncover<3->{
                (Q_{ent} - Q_{sai}) + (W_{ent} - W_{sai}) &
                = \Delta{E}_{sist} = (U_2 - U_1) + (E_{c,2} - E_{c,1}) + (E_{p,2} - E_{p,1})
                \rightharpoondown
            } \\[\medskipamount]
            \uncover<4->{
                \alert{Q_{liq,ent} - W_{liq,sai}} &
                \alert{= m\left[
                    (u_2 - u_1) +
                    (\mathbb{V}_2^2 - \mathbb{V}_1^2)/2 + 
                    g(\mathbb{Z}_2 - \mathbb{Z}_1)
                \right]}.
                \qquad\text{(expl.)}
            } \\[\medskipamount]
            \uncover<5->{
                \alert{Q - W} &
                \alert{= m\left[
                    (u_2 - u_1) +
                    (\mathbb{V}_2^2 - \mathbb{V}_1^2)/2 + 
                    g(\mathbb{Z}_2 - \mathbb{Z}_1)
                \right]}.
                \qquad\text{(impl.)}
            } %\\[\medskipamount]
        \end{align*}
