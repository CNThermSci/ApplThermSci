\subsection{Primeira Lei da Termodinâmica}
%-----------------------------------------------------------------------------------------------

    % !j 96 -i8
    %-------------------------------------------------------------------------------------------
    \begin{frame}{Enunciado}\vspace*{-2em}
        \uncover<1->{A \alert{1$^{\mathrm{a}}$ lei da termodinâmica} estabelece que:}
        \begin{itemize}
            \item<1-> \alert{Energia} é uma quantidade \alert{conservada}. \\[\bigskipamount]
        \end{itemize}
        \uncover<2->{Este \alert{princípio da conservação da energia}:}
        \begin{itemize}
            \item<2-> É exaustivamente \alert{confirmado} em experimentos.
        \end{itemize}
    \end{frame}
    %-------------------------------------------------------------------------------------------

    % !j 96 -i8
    %-------------------------------------------------------------------------------------------
    \begin{frame}{Implicações}\vspace*{-2em}
        \uncover<1->{Logo, no universo observável:}
        \begin{itemize}
            \item<1-> \alert{Não há processos físicos} que \alert{criem} energia,
                \\[\medskipamount]
            \item<2-> \alert{Nem processos físicos} que \alert{destruam} energia.
                \\[\medskipamount]
            \item<3-> \alert{Processos físicos} podem apenas \alert{converter} energia de uma
                forma a outra. \\[\medskipamount]
        \end{itemize}
        \uncover<4->{A Relatividade Especial de Einstein:}
        \begin{itemize}
            \item<4-> \alert{Unificou} as conservações de \alert{massa} e de \alert{energia};
                \\[\medskipamount]
            \item<5-> Através da \alert{equivalência massa-energia} expressa por $E_{eq} =
                c^2m$.  \\[\medskipamount]
            \item<6-> Portanto, a quantidade $E_{tot} = c^2m + E_{outras}$ é \alert{conservada}.
        \end{itemize}
    \end{frame}
